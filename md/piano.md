这段代码会自动为刚才 CSS 定义的 .piano-btn 绑定点击事件，并播放对应的钢琴音阶。

JavaScript
// 创建音频上下文
const audioCtx = new (window.AudioContext || window.webkitAudioContext)();

// 定义音阶频率 (C4 到 B5)
const frequencies = {
    '1̣': 261.63, '2̣': 293.66, '3̣': 329.63, '4̣': 349.23, '5̣': 392.00, '6̣': 440.00, '7̣': 493.88, // 低音
    '1': 523.25, '2': 587.33, '3': 659.25, '4': 698.46, '5': 783.99, '6': 880.00, '7': 987.77,  // 中音
    '1̇': 1046.50, '2̇': 1174.66, '3̇': 1318.51, '4̇': 1396.91, '5̇': 1567.98, '6̇': 1760.00, '7̇': 1975.53 // 高音
};

function playNote(frequency) {
    const oscillator = audioCtx.createOscillator();
    const gainNode = audioCtx.createGain();

    oscillator.type = 'sine'; // 模拟钢琴的圆润音色
    oscillator.frequency.setValueAtTime(frequency, audioCtx.currentTime);

    gainNode.gain.setValueAtTime(0.5, audioCtx.currentTime);
    gainNode.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 1); // 音频衰减效果

    oscillator.connect(gainNode);
    gainNode.connect(audioCtx.destination);

    oscillator.start();
    oscillator.stop(audioCtx.currentTime + 1);
}

// 绑定点击事件
document.querySelectorAll('.piano-btn').forEach(button => {
    button.addEventListener('mousedown', () => {
        const note = button.querySelector('span').innerText;
        if (frequencies[note]) {
            playNote(frequencies[note]);
            button.classList.add('active');
        }
    });
    button.addEventListener('mouseup', () => button.classList.remove('active'));
    button.addEventListener('mouseleave', () => button.classList.remove('active'));
});
